information_schema

information_schema 存储着所有的数据库的相关信息

SCHEMATA存储所有数据库名

select SCHEMA_NAME from SCHEMATA查询主要内容
schemata –> schema_name

TABLES存储所有的数据库表名

select TABLE_SCHEMA from TABLES查询主要内容
tables –> table_name –> table_schema

COLUMNS存储所有列名

select COLUMNS_NAME from COLUMNS查询主要内容
columns –> columns_name –> table_name
